        case $1 in
	    --get_boot_pd ) 
            _pd="boot"
	    _arg=$1
            ;;
            --get_boot_pd_part ) 
	    _pd="part"
	    _arg=$1
	    ;;
	    --boot )
	    _pd="rootfs2"
	    ;;
            *)
	    /sbin/hal_app-orig $*
	    exit
	esac
